API Introduction

Introduction

Mindband Connect for Cassia is template for a progressive web app. It includes mindband.js - a npm module

  • mindband.js is a javascript library. It offers connection to Cassia Hub and it receives Server Side Events from Cassia Hub.

      npm install --save mindband-js

  • mindband-connect-template is a Nuxt.js template

    • client is a sample application to demonstrate how to use mindband.js

    • server is a stub server. It returns a static response of server side events of scanning or mindband's data for testing instead of using a real Cassia Hub

      npm install --save mindband-connect-for-cassia

How to use mindband.js

you can find API usages in client/api/index.js. The basic calling sequence is like,

  1. scanning mindbands
    • MAC address of mindbands are recevied
  2. close the scan
  3. connecting a mindband
  4. starting Server Side Events
    • stream data of SSE is received
  5. close SSE
  6. close connection

Importing mindband.js 

import Mindband from '~/plugins/mindband'
var md = new Mindband(process.env.CASSIA_URL);

startScan & closeScan

  • {baddrs['34:81:F4:0E:BC:BE']}
export function startScan(filter, callback){
    var outS = kefir.fromEvents(md, "scanned")
    md.startScan(URL_SCAN)
    outS.onValue(x=>{
        var device = x.bdaddrs[0]
        callback(device)
      }
    )
}

export function closeScan(){
    md.closeScan();
}

startSSE & closeSSE

  • kefir - a reactive programming library to handle the data stream from mindbands.js
    • {'device':'34:81:F4:0E:BC:BE'}
    • {'signal': 0-255}
    • {'attention': 0-100}
    • {'meditation': 0-100}
    • {'wave': int16}
export function closeSSE(){
    md.closeSSE();
}

export function receiveSSE(callback){
    var outS = kefir.zip([
      asProp('device'),
      asProp('signal'),
      asProp('meditation'),
      asProp('attention'),
    ]).map(toObj)
    //
    md.startSSE();
    //
    outS.onValue(x=>{
        //console.log(x);
        callback(x)
      }
    )
}

function toObj (objs) {
  return objs.reduce(function (acc,o) {
    var k  = Object.keys(o)[0]
    acc[k] = o[k]
    return acc
  }, {})
}

function prop (p) {
  return (v) => {
    var r = {}
    r[p]  = v
    return r
  }
}

function asProp (ev) {
  return kefir.fromEvents(md, ev).map(prop(ev))
}
