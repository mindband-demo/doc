<script>
import * as API from '../api'
import Chart from 'chart.js'
let charts = {}
//
export default {
  methods: {
    startScan (e) {
      API.startScan(null, data =>{
        this.$store.commit('devices/push', data.bdaddr)
      })
    },
    closeScan (e) {
      API.closeScan()
    },
    connect(device){
      API.connect(device)
    },
    disconnect(device){
      API.disconnect(device)
    },
    receiveSSE(){
      API.receiveSSE(data=>{
        if (this.SSE[data.device]==null ){
          this.SSE[data.device] = {}
          var c1 = this.createChart(data.device)
          charts[data.device]= c1
        }
        if (this.SSE[data.device].messages==null ){
           this.SSE[data.device].messages = []
        }
        this.SSE[data.device].messages.push(data)
        var c1 = charts[data.device]
        if (c1){
          this.updateChart(c1, data)
        }
      })
    },
    closeSSE(){
      API.closeSSE()
    },
    createChart(id){
      console.log(id)
      let ctx = document.getElementById(id)
      if (ctx){
        let labels = []
        for (var i = 0; i < 60; i++) { 
          if ((i + 1) % 5 != 0) { 
            labels.push(""); 
          } else { 
            labels.push((i + 1) + "s"); 
          } 
        };
        //
        return Chart.Line(ctx, {
          data: {
            labels:labels,
            datasets: [{
              label: 'meditation',
              data: [],
              backgroundColor: 'rgba(255,255,102, 0.2)'
            },
            {
              label: 'attention',
              data: [],
              backgroundColor: 'rgba(0,255,102, 0.2)'
            }]
          },
          options: {
            responsive: true,
            scales: {
              xAxes: [{
                ticks: {
                  beginAtZero: true,
                  min: 0,
                  max: 100 
                }
              }],
              yAxes: [{
                ticks: {
                  beginAtZero: true,
                  min: 0,
                  max: 255
                }
              }]
            }
          }
        })
      }
    },
    updateChart(chart, data){
      chart.data.datasets.forEach((dataset) => {
        if (data.eeg){
           dataset.data.push({x:dataset.data.length, y:data.eeg[dataset.label]});
        }else{
           dataset.data.push({x:dataset.data.length, y:data[dataset.label]});
        }
      });
      chart.update();
    }
  },
}
