  //connection
  router.post('/gap/nodes/:id/connection', middleware, async (ctx, next) => {
    const id = ctx.params.id
    ctx.type = 'json'
    ctx.body = {
      message: 'connected',
      id:id
    }
  })

  //disconnect
  router.delete('/gap/nodes/:device/connection', middleware, async (ctx, next) => {
    const id = ctx.params.device
    ctx.type = 'json'
    ctx.body = {
      message: 'disconnected',
      id:id
    }
  })

  //services
  router.get('/gatt/nodes/:device/services', middleware, async (ctx, next) => {
    const id = ctx.params.device
    ctx.type = 'json'
    ctx.body = {
      message: 'services',
      id:id
    }
  })

//characteristics
  router.get('/gatt/nodes/:device/services/:ServiceUUID/characteristics', middleware, async (ctx, next) => {
    const id = ctx.params.device
    const uuid = ctx.params.ServiceUUID
    ctx.type = 'json'
    ctx.body = {
      message: 'characteristics',
      id:id,
      uuid:uuid
    }
  })

  //Notify
  router.get('/gatt/nodes/:device/handle/:hndl/value/:val', middleware, async (ctx, next) => {
    const id = ctx.params.device
    const Hn = ctx.params.hndl
    const Vl = ctx.params.val
    ctx.type = 'json'
    ctx.body = {
      message: "Notify",
      id:id,
      handle:Hn,
      value:Vl
    }
  })
